CREATE TABLE FNCTRANS 'Entry Journal' 1
CURDATE (DATE,8,'Refer. Date')
FNCDATE (DATE,8,'Due Date')
SUM1 (REAL,16,2,'Sum')
SUM3 (REAL,16,2,'Sum (Trans. Curr)')
EXCHANGE3 (REAL,17,9,'Exchange Rate')
SUMDEBIT (REAL,16,2,'Total Debit')
SUMCREDIT (REAL,16,2,'Total Credit')
IVNUM (CHAR,16,'Reference')
UDATE (DATE,14,'Time Stamp')
SUM2 (REAL,16,2,'Sum ({F.F})')
EXCHANGE2 (REAL,17,9,'Exch. Rate ({F.F})')
CFNCTRANS (INT,13,'Related Trans (ID)')
REF (INT,13,'Reference (ID)')
EXEC (INT,13,'Entity (ID)')
CHECKING (CHAR,1,'Checking')
BALDATE (DATE,8,'Transact Date')
STORNOFLAG (CHAR,1,'Reversed?')
BOOKNUM (CHAR,8,'Ext Entry Number')
LOADFNCENV (INT,13,'Source (ID)')
INDEXDATE (DATE,8,'Adjustment Date')
DETAILS (CHAR,24,'Details')
;
CREATE TABLE FNCPATTERNS 'Journal Entry Codes' 0
FNCPATNAME (CHAR,5,'Entry Code')
;
CREATE TABLE GENLEDGERS 'Define Fiscal Years' 0
GLNAME (CHAR,6,'Fiscal Year')
;
CREATE TABLE ACCOUNTS1 'Ledger Accounts' 1
ACCNAME1 (CHAR,16,'Account No.')
;

CREATE TABLE ACCOUNTS2 'Ledger Accounts' 1
ACCNAME2 (CHAR,16,'Account No.')
;
CREATE TABLE CURRENCIES 'Currencies' 0
CODE (CHAR,3,'Curr')
;


WHERE FNCTRANS.GL = GENLEDGERS.GL
AND FNCTRANS.ACCOUNT1 = ACCOUNTS1.ACCOUNT
AND FNCTRANS.ACCOUNT2 = ACCOUNTS2.ACCOUNT
AND FNCTRANS.FNCPATTERN = FNCPATTERNS.FNCPATTERN
AND FNCTRANS.CURRENCY3 = CURRENCIES.CURRENCY
AND FINAL = 'Y'
AND (ACCOUNT1 = :OLDACC OR ACCOUNT2 = :OLDACC)
